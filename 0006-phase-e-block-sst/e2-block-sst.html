STEP E2 · Phase E block-based SST

block-based SST + 稀疏索引

用 E1 的块拼出完整 SST。SstBuilder 把 entry 攒进当前块,超 BLOCK_SIZE 就封口成一个 data block、往稀疏索引记一条 (块末键, BlockHandle);finish 写出 [data blocks][index block][footer]。SstReader::open 只载入稀疏索引(小);get 二分索引定位唯一候选块、用 read_exact_at 只读那一块。读放大从 Phase D 的 O(整文件) 降到 O(一块)。

GOAL

重写 sst.rs:新增 BlockHandle{offset,len};SstBuilder 攒块 + 超阈值 flush_block + 稀疏索引;finish 写三段式文件;SstReader 持 File + 稀疏索引,get partition_point 定位候选块、read_block 只读一块。对外签名 与 Phase D 一致(new/add/finish/open/get),所以 engine.rs 一行不改。

Done when: 差分 vs BTreeMap(强制多块)绿;读放大不变量(一次点查 blocks_read 增量 ≤ 1)绿;跨块边界键(每块首/末键命中)绿;引擎旧测试全过。

三段式文件 + 稀疏索引(格式锚)

为什么索引项存"块的末键"? 稀疏索引每块只留一个 key 当"路标"。存末键(块内最大 key)后,查找就是"找第一个末键 ≥ target 的块"——因为块按 key range 不重叠且升序,target 若存在,必在这一块里(它 ≤ 该块末键,且 > 前一块末键)。这正是 RocksDB IndexBuilder::AddIndexEntry(last_key_in_current_block, ...) 的语义。

代码 1 — 引入 + BlockHandle

use crate::block::{Block, BlockBuilder};
use anyhow::{Result, bail};
use bytes::Bytes;
use std::fs::File;
use std::io::Write;
use std::os::unix::fs::FileExt; // read_exact_at:&self 即可按偏移随机读(免整文件载入)
use std::path::Path;
use std::sync::atomic::{AtomicU64, Ordering};

const MAGIC: u64 = 0x6d696e695f737374; // "mini_sst"
const FOOTER_LEN: usize = 24;
const BLOCK_SIZE: usize = 4096; // 默认块阈值 ~4KB;测试用 with_block_size 调小强制多块

/// 定位一个块在文件里的位置:偏移 + 长度。RocksDB 的 BlockHandle 同义。
#[derive(Clone, Copy, Debug)]
pub struct BlockHandle { pub offset: u64, pub len: u64 }
impl BlockHandle {
    fn encode(&self, buf: &mut Vec<u8>) {
        buf.extend_from_slice(&self.offset.to_le_bytes());
        buf.extend_from_slice(&self.len.to_le_bytes());
    }
    fn decode(b: &[u8]) -> Self {
        Self {
            offset: u64::from_le_bytes(b[0..8].try_into().unwrap()),
            len: u64::from_le_bytes(b[8..16].try_into().unwrap()),
        }
    }
}

代码 2 — SstBuilder(攒块 + 稀疏索引)

pub struct SstBuilder {
    data: Vec<u8>,                    // 已封口的 data blocks(累积)
    block: BlockBuilder,              // 当前正在攒的块
    index: Vec<(Bytes, BlockHandle)>, // 稀疏索引:每块一条 (末键, handle)
    last_key: Option<Bytes>,          // 升序校验 + 当前块的末键
    block_size: usize,
}
impl SstBuilder {
    pub fn new() -> Self { Self::with_block_size(BLOCK_SIZE) }
    pub fn with_block_size(block_size: usize) -> Self {
        Self { data: Vec::new(), block: BlockBuilder::new(), index: Vec::new(), last_key: None, block_size }
    }

    pub fn add(&mut self, key: &[u8], val: &[u8]) -> Result<()> {
        if let Some(last) = &self.last_key && key <= last.as_ref() {
            bail!("SstBuilder.add 非升序: {:?} <= {:?}", key, last);
        }
        self.block.add(key, val);
        self.last_key = Some(Bytes::copy_from_slice(key));
        if self.block.estimated_size() >= self.block_size { self.flush_block(); } // 块满 → 封口
        Ok(())
    }

    /// 把当前块封口写入 data 区,并把(末键, handle)记进稀疏索引。空块跳过。
    fn flush_block(&mut self) {
        if self.block.is_empty() { return; }
        let block = std::mem::take(&mut self.block).finish();
        let offset = self.data.len() as u64;
        let len = block.len() as u64;
        self.data.extend_from_slice(&block);
        let last = self.last_key.clone().unwrap(); // 此刻 last_key 正是本块末键
        self.index.push((last, BlockHandle { offset, len }));
    }

    pub fn finish(mut self, path: &Path) -> Result<()> {
        self.flush_block(); // 收尾:最后一个未满块也封口
        let mut f = File::create(path)?;
        f.write_all(&self.data)?;
        let index_offset = self.data.len() as u64;
        let mut ib = BlockBuilder::new();   // 索引块复用 BlockBuilder
        let mut hb = Vec::with_capacity(16);
        for (k, h) in &self.index { hb.clear(); h.encode(&mut hb); ib.add(k, &hb); }
        let index_block = ib.finish();
        f.write_all(&index_block)?;
        let mut footer = Vec::with_capacity(FOOTER_LEN);
        footer.extend_from_slice(&index_offset.to_le_bytes());
        footer.extend_from_slice(&(index_block.len() as u64).to_le_bytes());
        footer.extend_from_slice(&MAGIC.to_le_bytes());
        f.write_all(&footer)?;
        f.sync_all()?;
        Ok(())
    }
}
impl Default for SstBuilder { fn default() -> Self { Self::new() } }
std::mem::take 是封块的关键招. flush_block 要"取走当前 builder、换个空的继续攒"。std::mem::take(&mut self.block) 用 Default 造一个空 BlockBuilder 顶上去、把旧的move 出来 finish 掉——这就是为什么 E1 给 BlockBuilder 实现了 Default。无需 Option 包裹、无需 clone。

代码 3 — SstReader(只读候选块)

/// 只在 open 时载入小小的稀疏索引;data block 在 get 时按需读取。
pub struct SstReader {
    file: File,
    index: Vec<(Bytes, BlockHandle)>, // 升序的 (块末键, handle)
    blocks_read: AtomicU64,           // 仪表:累计读了多少 data block(读放大佐证)
}
impl SstReader {
    pub fn open(path: &Path) -> Result<Self> {
        let file = File::open(path)?;
        let file_len = file.metadata()?.len();
        if file_len < FOOTER_LEN as u64 { bail!("SST 太小,无 footer"); }
        let mut footer = [0u8; FOOTER_LEN];
        file.read_exact_at(&mut footer, file_len - FOOTER_LEN as u64)?;
        let index_offset = u64::from_le_bytes(footer[0..8].try_into()?);
        let index_len = u64::from_le_bytes(footer[8..16].try_into()?);
        let magic = u64::from_le_bytes(footer[16..24].try_into()?);
        if magic != MAGIC { bail!("magic 不符,非本格式 SST"); }
        let mut ibuf = vec![0u8; index_len as usize]; // 只读索引块,不读整文件
        file.read_exact_at(&mut ibuf, index_offset)?;
        let index_block = Block::decode(Bytes::from(ibuf));
        let mut index = Vec::with_capacity(index_block.len());
        for i in 0..index_block.len() {
            let (k, v) = index_block.entry_at(i);
            index.push((Bytes::copy_from_slice(k), BlockHandle::decode(v)));
        }
        Ok(Self { file, index, blocks_read: AtomicU64::new(0) })
    }

    fn read_block(&self, h: BlockHandle) -> Result<Block> {
        let mut buf = vec![0u8; h.len as usize];
        self.file.read_exact_at(&mut buf, h.offset)?;
        self.blocks_read.fetch_add(1, Ordering::Relaxed);
        Ok(Block::decode(Bytes::from(buf)))
    }

    /// 点查:二分稀疏索引找唯一候选块(末键 ≥ key 的第一块),只读那一块,块内再二分。
    pub fn get(&self, key: &[u8]) -> Option<Bytes> {
        let idx = self.index.partition_point(|(last, _)| last.as_ref() < key);
        let (_, handle) = self.index.get(idx)?; // idx 越界 = key 大于所有末键 → miss,不读块
        let block = self.read_block(*handle).ok()?;
        block.get(key).map(Bytes::copy_from_slice) // 点查边界拷出;零拷贝迭代在 Phase F
    }

    pub fn num_blocks(&self) -> usize { self.index.len() }
    pub fn blocks_read(&self) -> u64 { self.blocks_read.load(Ordering::Relaxed) }
}

engine.rs 调的还是 SstBuilder::new/add/finish 和 SstReader::open/get——签名全没变,所以引擎一行不改。块化彻底封装在 SST 模块边界内,这是"深模块、窄接口"的实证。

🧰 Coding style:newtype 句柄 · partition_point(lower_bound)· 仪表化 &self

  • newtype 句柄(BlockHandle):不用裸 (u64,u64) 元组到处传,而是给"块的位置"一个有名字的类型。它把 offset/len 绑成一个不可拆的概念,encode/decode 收在一处,误传两个无关的 u64 直接编译不过。这是"为领域概念造类型"的范式(对比 Phase A 的 Key=Bytes 别名——句柄更进一步,有自己的方法)。
  • partition_point = lower_bound:index.partition_point(|(last,_)| last < key) 返回第一个不满足谓词的下标,即"第一个末键 ≥ key 的块"。这正是 C++ std::lower_bound。比手写二分边界(±1 off-by-one 重灾区)安全得多。记住这个映射:"找第一个 ≥ x" = partition_point(< x)。
  • 仪表化但保持 &self:blocks_read 用 AtomicU64 而非 u64,所以 read_block 能在 &self(共享只读)下自增计数——读路径不需要 &mut。这既服务测试(读放大断言),也预演了真引擎里 &self 并发读 + 原子统计计数器(命中率、IO 次数)的常见形态。"在不可变接口下做可变统计",原子量是标准答案。

🧠 System mindset:稀疏索引 · 读放大 · 按需 IO 而非全载

这一步真正改变的是读的成本结构。两件事值得拎出来跨域看:

概念本阶段怎么做 / 它在别处长什么样
稀疏索引每块只留一个路标(末键),索引大小 = O(块数) 而非 O(条数)。同构于:B-tree 内部节点只存分隔键不存数据、数据库的稀疏索引(每数据页一个索引项)、Kafka 的 .index(每若干条消息一个 offset→物理位置)、视频文件的关键帧索引。
读放大为读一个 key 实际搬了多少字节。Phase D:整文件;Phase E:一块(~4KB)。LSM 三大放大(读/写/空间)里,读放大是这条线的主角;后续 bloom(Phase 后期)让 miss 连块都不读、block cache(Phase S)让热块不重复读盘。
按需 IO(read_exact_at)不再 read_to_end 把整文件吸进内存,而是按 (offset,len) 随机读。这就是 pread(2);真引擎更进一步用 mmap 或 io_uring。"只为需要的字节付 IO"是所有大文件随机访问系统(数据库、视频点播 seek、分页加载)的共同信条。

迁移收获:稀疏索引是一个空间/读放大的对冲。索引越稀疏(块越大)→ 索引越省内存,但每次点查搬的块越大。"用一个常驻内存的小索引,把一次大文件随机查,变成一次小块 IO"——这个模式在你做任何"大文件 + 随机查"的系统时都会重演。

🧪 Harness 系统:读放大不变量(性能契约可执行化)

为什么这样验

差分(vs BTreeMap)只能证明"结果对",证明不了"读得省"——一个把整文件载入再二分的实现,差分照样全绿。但 Phase E 的全部价值就在"读得省"。于是我们给 SstReader 埋一个 blocks_read 计数器,把性能契约钉成断言:无论文件被切成多少块,一次点查 blocks_read 的增量最多是 1;查一个大于所有键的 miss,增量是 0(索引直接判空,一块都不碰)。这类"断言资源消耗量级"的 harness(数 IO 次数 / 分配次数 / SQL 查询数),是把"快"从口号变成回归测试的关键——它能抓住"功能没坏但悄悄退化成全表扫描"这种最阴险的性能回归(ORM 的 N+1 查询就是被这种测试抓的)。

    // 读放大不变量:无论多少块,一次点查最多读一个 data block。
    #[test]
    fn point_get_reads_one_block() {
        let dir = tempdir().unwrap();
        let path = dir.path().join("amp.sst");
        let mut b = SstBuilder::with_block_size(16); // 小阈值 → 多块
        for i in 0..200u32 { b.add(format!("k{i:04}").as_bytes(), b"payload-data").unwrap(); }
        b.finish(&path).unwrap();
        let r = SstReader::open(&path).unwrap();
        assert!(r.num_blocks() > 1, "应被切成多块");

        r.get(b"k0100").unwrap(); assert_eq!(r.blocks_read(), 1); // 命中:读 1 块
        r.get(b"k0150").unwrap(); assert_eq!(r.blocks_read(), 2); // 再 1 块(累计 2)
        r.get(b"k0100x");          assert_eq!(r.blocks_read(), 3); // miss(块内不存在):仍读 1 块
        r.get(b"zzzz");            assert_eq!(r.blocks_read(), 3); // miss(越界):一块都不读
    }

    // 边界:逐块首键/末键都要命中(稀疏索引 partition_point 的临界点)。
    #[test]
    fn cross_block_boundary_keys() {
        let dir = tempdir().unwrap();
        let path = dir.path().join("bnd.sst");
        let mut b = SstBuilder::with_block_size(40); // 每块约 3 条
        let keys: Vec<String> = (0..30).map(|i| format!("k{i:03}")).collect();
        for k in &keys { b.add(k.as_bytes(), b"v").unwrap(); }
        b.finish(&path).unwrap();
        let r = SstReader::open(&path).unwrap();
        assert!(r.num_blocks() > 1);
        for k in &keys { assert_eq!(r.get(k.as_bytes()).as_deref(), Some(&b"v"[..]), "{k}"); }
    }

    // 差分:随机有序输入,SST.get == BTreeMap.get(强制多块)。
    #[test]
    fn matches_btreemap_oracle() {
        let dir = tempdir().unwrap();
        let path = dir.path().join("o.sst");
        let mut model = BTreeMap::new();
        for i in 0..500u32 { model.insert(format!("key{:05}", i * 7 % 1000), format!("val{}", i)); }
        let mut b = SstBuilder::with_block_size(64); // 多块
        for (k, v) in &model { b.add(k.as_bytes(), v.as_bytes()).unwrap(); }
        b.finish(&path).unwrap();
        let r = SstReader::open(&path).unwrap();
        assert!(r.num_blocks() > 1);
        for (k, v) in &model { assert_eq!(r.get(k.as_bytes()).as_deref(), Some(v.as_bytes())); }
        assert_eq!(r.get(b"nope"), None);
    }

核心问题(自己答)

E2 想透这几个
  1. miss 仍要读一块:查一个落在某块 key range 内、但实际不存在的 key(如 k0100x),我们仍读了一整块才发现没有。一个 SST 上百万次 miss 点查,这是巨大浪费。哪个结构能让我们"读块之前"就以高概率否定?(提示:per-SST 或 per-block 的 bloom filter。它怎么做到"说没有就一定没有、说有可能误报"?)
  2. 稀疏索引常驻 vs 多级索引:open 把整个稀疏索引读进内存常驻。一个 1TB 的 SST、4KB 块,有 ~2.5 亿块,索引本身就几个 GB——还能全常驻吗?RocksDB 的 partitioned index 怎么解?(提示:把索引本身也切成块,再加一层顶层索引——索引的索引。这和我们 data block 的关系,是不是很像?)
  3. 每次 get 重读块的浪费:同一个热块被 1000 次点查命中,我们 read_exact_at 了 1000 次(虽然 OS page cache 可能挡掉真实磁盘 IO)。blocks_read 会数到 1000。哪个组件该把解析好的 Block 缓存在用户态,省掉重复 decode?(Phase S:block cache,一个 LRU。想想它的 key 该是什么——文件号 + 块 offset?)
  4. 点查拷出 vs 迭代零拷贝:SstReader::get 最后 Bytes::copy_from_slice 把 value 拷出来了——因为块是 get 内的局部变量,函数返回时块就 drop 了,借入它的切片活不过函数边界。E1 的零拷贝在这里没兑现。Phase F 的迭代器怎么改变这一点?(提示:迭代器持有当前块,只要迭代器活着,块就活着,value() -> &[u8] 就能一路借下去。零拷贝的收益要在批量迭代时才真正显现,而非单次点查。)

下一步:E3 — 基准 flat vs block 的 A/B. 块化"快了多少、省了多少"还只是声称。E3 把 Phase D 扁平 SST 和现在的块化 SST 放进同一个二进制,同种子同负载实测磁盘 / 索引内存 / 冷热读延迟(对标 db_bench)。你会看到块化在磁盘(~15%)+ 索引内存(24×)+ 冷选择性读(可达 33×)上赢,却在热点查上略输——这条诚实的对照,正好给出 E4(前缀压缩)、E5(块尾 CRC+压缩)以及更远的 Phase S(block cache)的动机。再往后 Phase F 用迭代器兑现 E1 埋下的零拷贝。
